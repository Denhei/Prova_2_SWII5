create database sales

use sales

create table salesman(
SALESMAN_ID INT auto_increment,
NAME VARCHAR(30),
CITY VARCHAR(15),
COMISSION DECIMAL(5,2),
CONSTRAINT PK_SALESMAN PRIMARY KEY (SALESMAN_ID)
)

create table customer(
CUSTOMER_ID INT auto_increment,
CUST_NAME VARCHAR(30),
CITY VARCHAR(15),
GRADE INT,
SALESMAN_ID INT,
CONSTRAINT PK_CUSTOMER PRIMARY KEY (CUSTOMER_ID)
)

create table orders(
ORD_NO INT AUTO_INCREMENT,
PURCH_AMT DECIMAL(8,2),
ORD_DATE DATE,
CUSTOMER_ID INT,
SALESMAN_ID INT,
CONSTRAINT  PK_ORDERS PRIMARY KEY (ORD_NO),
CONSTRAINT FK_ORDERS_SALESMAN FOREIGN KEY (SALESMAN_ID)
	REFERENCES SALESMAN (SALESMAN_ID),
CONSTRAINT FK_ORDERS_CUSTOMER FOREIGN KEY (CUSTOMER_ID)
	REFERENCES CUSTOMER (CUSTOMER_ID)
)
